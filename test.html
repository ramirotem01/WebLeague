// ... (שאר הקוד נשאר ללא שינוי עד לפונקציה renderTable) function renderTable(data) { const tbody = document.getElementById('leaderboardBody'); let html = ''; data.forEach((item, i) => { // שורת נתונים רגילה html += ` ${i + 1}
${item.title}
PR ${item.pr}
${Object.entries(item.metrics || {}).filter(([k,v]) => v > 0).slice(0, 3).map(([k,v]) => ` ${k} `).join('')}
${item.votes}
`; // פרסומת דקה (מופיעה כל 10 שורות) if ((i + 1) % 10 === 0 && i !== data.length - 1) { html += `
Ad Exclusive Deals for Developers
Visit Now
`; } }); tbody.innerHTML = html; }
