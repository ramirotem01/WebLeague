window.analyzeWebsite = async function() { const urlInput = document.getElementById('urlInput'); let url = urlInput.value.trim(); if(!url.startsWith('http')) url = 'https://' + url; document.getElementById('btnLabel').innerText = "Scanning..."; document.getElementById('btnSpinner').classList.remove('hidden'); try { const raw = await robustProxyFetch(url); if (!raw) throw new Error("Proxy block"); const docParsed = new DOMParser().parseFromString(raw, 'text/html'); const domainName = url.split('/')[2].replace('www.', '').split('.')[0]; // --- היררכיית חילוץ תיאור משופרת --- const metaDesc = docParsed.querySelector('meta[property="og:description"]')?.content || docParsed.querySelector('meta[name="description"]')?.content || docParsed.querySelector('meta[name="twitter:description"]')?.content || docParsed.querySelector('p')?.innerText || // מנסה לקחת פסקה ראשונה מהדף `Explore ${domainName} - A powerful AI solution listed on pcs365.`; const metaTitle = docParsed.querySelector('title')?.innerText || docParsed.querySelector('meta[property="og:title"]')?.content || domainName; const finalTitle = metaTitle.split('|')[0].split('-')[0].trim(); const finalDesc = metaDesc.trim().substring(0, 160).replace(/\s+/g, ' ') + (metaDesc.length > 160 ? '...' : ''); const s = await getDeepSignals(domainName, url); window.tempAI = { url, title: finalTitle, description: finalDesc, pr: s.pr, icon: `https://www.google.com/s2/favicons?sz=128&domain=${url}`, votes: 1, metrics: { reddit: s.r, hn: s.h, github: s.g } }; updatePreview(window.tempAI); } catch (e) { console.error(e); alert("Note: This site is protecting its data. You can still list it, but some metadata might be limited."); } finally { document.getElementById('btnLabel').innerText = "Analyze & List"; document.getElementById('btnSpinner').classList.add('hidden'); } };
